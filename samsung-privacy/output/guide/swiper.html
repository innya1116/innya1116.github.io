Swiper

[html]
  • .swiper > .swiper-container > .swiper-wrapper > .swiper-silde 구조를 지켜야 swiper 플러그인 적용됨
  • swiper-pagination, swiper-autoplay 의 위치는 디자인에 따라서 변경해도 무방

autoplay

<div class="swiper" data-options='{
  "loop": false,
  "autoplay": "2000",
  "pagination" : "bullet"
  }'>
  <div class="swiper-container">
    <div class="swiper-wrapper">
      <!-- Slides -->
      <div class="swiper-slide">
        slide contents
      </div>
      <div class="swiper-slide">
        slide contents
      </div>
      <div class="swiper-slide">
        slide contents
      </div>
      <!--// Slides -->
    </div>
  </div>
  <div class="swiper-btn-wrap">
    <div class="swiper-pagination"></div>
    <div class="swiper-autoplay">
      <button type="button" class="swiper-button-autoplay--pause on"><span class="hidden">정지</span>
        <svg class="icon" focusable="false">
          <use href="../images/sprite-svg.svg#swiper-pause"></use>
        </svg>
      </button>
      <button type="button" class="swiper-button-autoplay--play"><span class="hidden">재생</span>
        <svg class="icon" focusable="false">
          <use href="../images/sprite-svg.svg#swiper-play"></use>
        </svg>
      </button>
    </div>
  </div>
</div>
