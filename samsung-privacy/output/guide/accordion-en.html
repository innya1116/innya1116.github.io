Accordion

Usage

  • If the default is closed, change the text 'aria-expanded="false" button.accordion__btn > span close' in button.accordion__btn, and apply 'aria-hidden="true" style="height: 0;"' to div.accordion__panel
  • Apply accordion by adding data-options to .acccordion 
      data-options='{
    "id": "accordion-id", //add id for creation toggle button and creation id/aria attribute js to panel
    "openType": "multi", //"single" .accordion-item single open / "multi" .accordion-item multiple open
    "animation": true, //Whether to use slide down motion(true/false)
    "animationSpeed": 400 //slide down speed
  }'

<div class="accordion" data-options='{
    "id": "privacyPolicy",
    "openType" : "multi",
    "animation": true,
    "animationSpeed": 0.4
  }'>
  <!-- Default Open Item -->
  <div class="accordion-item">
    <div class="accordion-header">
      <h2>Default Open Item</h2>
      <!-- (dev) Activate accordion button, aria-expanded="true" -->
      <button type="button" class="accordion-btn" aria-expanded="true">
        <!-- (dev) Activate accordion button, <span data-text='{"text":"open|close", "show":2}'>close</span> -->
        <span data-text='{"text":"open|close", "show":2}'>close</span>
      </button>
    </div>
    <!-- (dev) Activate accordion contents, aria-hidden="false" -->
    <div class="accordion-panel" aria-hidden="false">
      Item Contents
    </div>
  </div>
  <!-- Default Closed Item -->
  <div class="accordion-item">
    <div class="accordion-header">
      <h2>Default Item</h2>
      <button type="button" class="accordion-btn" aria-expanded="false"><span data-text='{"text":"open|close", "show":1}'>open</span></button>
    </div>
    <div class="accordion-panel" aria-hidden="false">
      Item Contents
    </div>
  </div>
  <div class="accordion-item">
