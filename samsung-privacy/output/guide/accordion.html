Accordion

Usage

  • Default가 닫힘일 경우 button.accordion__btn에 aria-expanded="false" button.accordion__btn>span close 텍스트 변경, div.accordion__panel에 aria-hidden="true" style="height: 0;" 적용
  • .accordion에 data-options 추가하여 accordion 응용 
      data-options='{
    "id": "accordion-id", //toggle 버튼과 panel에  id/aria 속성 js에서 생성을 위한 id 추가
    "openType": "multi", //"single" .accordion-item 한개 오픈 / "multi" .accordion-item 다중 오픈
    "animation": true, //슬라이드 다운 모션 적용 여부
    "animationSpeed": 400 //슬라이드 다운 스피드
  }'

<div class="accordion" data-options='{
    "id": "privacyPolicy",
    "openType" : "multi",
    "animation": true,
    "animationSpeed": 0.4
  }'>
  <!-- Default Open Item -->
  <div class="accordion-item">
    <div class="accordion-header">
      <h2>Default Open Item</h2>
      <!-- (dev) 아코디언 버튼 활성화 처리, aria-expanded="true" -->
      <button type="button" class="accordion-btn" aria-expanded="true">
        <!-- (dev) 아코디언 버튼 활성화 처리, <span data-text='{"text":"open|close", "show":2}'>close</span> -->
        <span data-text='{"text":"open|close", "show":2}'>close</span>
      </button>
    </div>
    <!-- (dev) 아코디언 콘텐츠 활성화 처리, aria-hidden="false" -->
    <div class="accordion-panel" aria-hidden="false">
      Item Contents
    </div>
  </div>
  <!-- Default Closed Item -->
  <div class="accordion-item">
    <div class="accordion-header">
      <h2>Default Item</h2>
      <button type="button" class="accordion-btn" aria-expanded="false"><span data-text='{"text":"open|close", "show":1}'>open</span></button>
    </div>
    <div class="accordion-panel" aria-hidden="false">
      Item Contents
    </div>
  </div>
  <div class="accordion-item">
    <div class="accordion-header">
      <h2>Default Item</h2>
      <button type="button" class="accordion-btn" aria-expanded="false"><span data-text='{"text":"open|close", "show":1}'>open</span></button>
    </div>
    <div class="accordion-panel" aria-hidden="false">
      Item Contents
    </div>
  </div>
</div><!--// .accordion -->
