Swiper

[html]
  • The swiper plug-in is applied only when the structure of div.swiper > div.swiper-container > div.swiper-wrapper > div.swiper-silde is observed.
  • The location of swiper-pagination and swiper-autoplay can be changed depending on the design.

autoplay

<div class="swiper" data-options='{
  "loop": false,
  "autoplay": "2000",
  "pagination" : "bullet"
  }'>
  <div class="swiper-container">
    <div class="swiper-wrapper">
      <!-- Slides -->
      <div class="swiper-slide">
        slide contents
      </div>
      <div class="swiper-slide">
        slide contents
      </div>
      <div class="swiper-slide">
        slide contents
      </div>
      <!--// Slides -->
    </div>
  </div>
  <div class="swiper-btn-wrap">
    <div class="swiper-pagination"></div>
    <div class="swiper-autoplay">
      <button type="button" class="swiper-button-autoplay--pause on"><span class="hidden">Stop</span>
        <svg class="icon" focusable="false">
          <use href="../images/sprite-svg.svg#swiper-pause"></use>
        </svg>
      </button>
      <button type="button" class="swiper-button-autoplay--play"><span class="hidden">Play</span>
        <svg class="icon" focusable="false">
          <use href="../images/sprite-svg.svg#swiper-play"></use>
        </svg>
      </button>
    </div>
  </div>
</div>
